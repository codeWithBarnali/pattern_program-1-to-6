    1    
   121   
  12321  
 1234321 
123454321


#include <stdio.h>

int main()
{
    int i,j,k,n;
    
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        k=1;
        for(j=1;j<2*n;j++)
        {
            if(j>=n+1-i && j<=n-1+i)
            {
                printf("%d", k);
                j<n ? k++ : k-- ;
            }
            else

            printf(" ");
            
        }
        printf("\n");
    }

    return 0;
}
