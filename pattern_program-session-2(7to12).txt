#include <stdio.h>

int main()
{
    
    int i,j,k,n;
    scanf("%d",&n);
    k=n;
    for(i=1;i<=n;i++)
    {   
        for(j=1;j<2*n;j++)
        {   
            if(j>=i  && j<=2*n-i)
            {   
                printf("%d", k);
                
                j<n ? k-- :++k;
                
            }
            else
            
            printf(" ");
            
        }
        
        k=k-2;
        printf("\n");
    }
    
    return 0;
}
